SELECT  order_id,
        created_at,
        status,
        num_of_item,
        gender
  
  FROM  bigquery-public-data.thelook_ecommerce.orders
  WHERE num_of_item > 2
        AND (created_at BETWEEN '2022-08-01 00:00:00 UTC' AND '2022-08-31 23:59:59 UTC')
        AND gender = 'F'
  
  LIMIT 10;
